\section{PERMUTATION planning problem}
\begin{verbatim}
(define (problem PERMUTATION)
   (:domain PRODUCTION)
   (:requirements :typing)
   (:objects zoneA zoneB - zone
             table - table
             engine - metal
             hood - metal)
   (:init (at-gripper zoneA)
          (free right)
          (at engine zoneB)
          (at hood zoneA))
   (:goal (at engine zoneA)
          (at hood zoneB)))
\end{verbatim}
